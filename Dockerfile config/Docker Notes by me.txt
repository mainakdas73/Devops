Put Dockerfile and source code file in same directory and build the image after which run and start the image as shown below.

>> Dockerfile

FROM ubuntu:latest

WORKDIR /app

RUN apt update -y 

RUN apt install apache2 -y

COPY . /var/www/html

EXPOSE 80

CMD ["apachectl", "-D", "FOREGROUND"]
